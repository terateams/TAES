```sql
  COUNT(*) FILTER (WHERE latency_ms < 200) * 100.0 / COUNT(*) as health_rate
FROM connection_events
WHERE timestamp >= NOW() - INTERVAL '1 hour'
  AND event_type = 'health_check'
GROUP BY workplace;
```
